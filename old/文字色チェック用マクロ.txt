Sub CheckSelectedColorAndBackgroundRGB()
    If Selection.Type = wdSelectionIP Then
        MsgBox "色を調べたい文字を選択してから実行してください。"
        Exit Sub
    End If
    
    Dim fontColor As Long, bgColor As Long
    Dim r As Long, g As Long, b As Long
    
    ' 文字色
    fontColor = Selection.Font.Color
    
    ' 背景色（Shading.BackgroundPatternColor）
    bgColor = Selection.Font.Shading.BackgroundPatternColor
    
    ' 文字色のRGB算出
    r = fontColor Mod 256
    g = (fontColor \ 256) Mod 256
    b = (fontColor \ 65536) Mod 256
    Dim msg As String
    msg = "【文字色 (Font)】" & vbCrLf & "R=" & r & ", G=" & g & ", B=" & b & vbCrLf & vbCrLf
    
    ' 背景色のRGB算出
    If bgColor = wdColorAutomatic Or bgColor = -16777216 Then
        msg = msg & "【背景色 (Back)】" & vbCrLf & "なし (透明)"
    Else
        r = bgColor Mod 256
        g = (bgColor \ 256) Mod 256
        b = (bgColor \ 65536) Mod 256
        msg = msg & "【背景色 (Back)】" & vbCrLf & "R=" & r & ", G=" & g & ", B=" & b
    End If
    
    MsgBox msg
End Sub
